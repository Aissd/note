```
var makeSound = function(animal) {
    if (animal && typeof animal.sound === 'function') {
        animal.sound();
    }
}

function Duck() {}

Duck.prototype.sound = function() {
    console.log('嘎嘎嘎');
}

function Chicken() {}

Chicken.prototype.sound = function() {
    console.log('咯咯咯');
}

var duck = new Duck();
makeSound(duck);

var chicken = new Chicken();
makeSound(chicken);
```
