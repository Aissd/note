### 多态背后的思想是将“不变的事物”与“可能改变的事物”分离开来
```
// “不变的事物”
var makeSound = function(animal) {
    if (animal && typeof animal.sound === 'function') {
        animal.sound();
    }
}

// “可能改变的事物”
function Duck() {}
Duck.prototype.sound = function() {
    console.log('嘎嘎嘎');
}

function Chicken() {}
Chicken.prototype.sound = function() {
    console.log('咯咯咯');
}

makeSound(new Duck());
makeSound(new Chicken());
```
